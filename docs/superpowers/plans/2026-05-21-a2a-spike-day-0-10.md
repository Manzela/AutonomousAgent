```python
"""A2A receiver — JSON-RPC dispatch over HTTP POST /.

Day-2 scope: parse JSON-RPC envelope, dispatch by `method`, return a
hard-coded synthetic Task for `message/send`. Auth is added Day 5;
telemetry Day 6; bridge to TaskSpec Day 7.

Spec: A2A v1.0.0 §3 (JSON-RPC envelope), §4.1 (message/send), §10 (errors).
"""

from __future__ import annotations

import json
from typing import Any

import ulid
from fastapi import FastAPI, Request, Response
from fastapi.responses import JSONResponse

app = FastAPI(title="Hermes A2A receiver", version="0.1.0-spike")


def _jsonrpc_error(req_id: Any, code: int, message: str) -> dict:
    return {"jsonrpc": "2.0", "id": req_id, "error": {"code": code, "message": message}}


def _jsonrpc_result(req_id: Any, result: Any) -> dict:
    return {"jsonrpc": "2.0", "id": req_id, "result": result}


def _handle_send_message(params: dict) -> dict:
    """Day-2 stub — returns a synthetic Task. Day 7 wires this to TaskSpec."""
    task_id = f"task-{ulid.new()}"
    return {
        "id": task_id,
        "status": {"state": "SUBMITTED"},
        "message": params.get("message", {}),
    }


_METHODS = {
    "message/send": _handle_send_message,
}


@app.post("/")
async def dispatch(request: Request) -> Response:
    raw = await request.body()
    try:
        envelope = json.loads(raw)
    except json.JSONDecodeError:
        return JSONResponse(_jsonrpc_error(None, -32700, "Parse error"))

    req_id = envelope.get("id")
    method = envelope.get("method")
    params = envelope.get("params", {})

    handler = _METHODS.get(method)
    if handler is None:
        return JSONResponse(_jsonrpc_error(req_id, -32601, f"Method not found: {method}"))

    try:
        result = handler(params)
    except Exception as exc:  # noqa: BLE001 — spike dispatcher reports any handler error
        return JSONResponse(_jsonrpc_error(req_id, -32603, f"Internal error: {exc}"))

    return JSONResponse(_jsonrpc_result(req_id, result))
```

```python
"""A2A sender — async client over httpx.

Day-3 scope: send_message round-trip; map JSON-RPC error codes to
typed Python exceptions. Auth (Day 5), telemetry (Day 6), streaming
(Day 4) layer on top.
"""

from __future__ import annotations

from typing import Any

import httpx
import ulid


class A2AError(Exception):
    """Base for all A2A wire errors."""

    code: int = 0


class A2AParseError(A2AError):
    code = -32700


class A2AInvalidRequest(A2AError):
    code = -32600


class A2AMethodNotFound(A2AError):
    code = -32601


class A2AInternalError(A2AError):
    code = -32603


_CODE_TO_EXC: dict[int, type[A2AError]] = {
    -32700: A2AParseError,
    -32600: A2AInvalidRequest,
    -32601: A2AMethodNotFound,
    -32603: A2AInternalError,
}


class A2AClient:
    """Async A2A client. Pass `_transport` (a TestClient) for in-process testing."""

    def __init__(self, peer_url: str, _transport: Any | None = None) -> None:
        self.peer_url = peer_url
        self._transport = _transport

    async def _call(self, method: str, params: dict) -> Any:
        envelope = {
            "jsonrpc": "2.0",
            "id": str(ulid.new()),
            "method": method,
            "params": params,
        }
        if self._transport is not None:
            # In-process: use TestClient synchronously
            resp = self._transport.post("/", json=envelope)
            body = resp.json()
        else:
            async with httpx.AsyncClient(timeout=10) as h:
                r = await h.post(self.peer_url, json=envelope)
                r.raise_for_status()
                body = r.json()

        if "error" in body:
            code = body["error"]["code"]
            exc = _CODE_TO_EXC.get(code, A2AError)
            raise exc(body["error"].get("message", ""))
        return body["result"]

    async def send_message(self, text: str) -> dict:
        return await self._call(
            "message/send",
            {"message": {"role": "USER", "parts": [{"text": text}]}},
        )
```

```python
from __future__ import annotations

import argparse
import asyncio
import json

from lib.a2a.client import A2AClient

    p = argparse.ArgumentParser(prog="a2a-cli")

        return 0
    return 1


if __name__ == "__main__":
```

```python
"""A2A auth — JWT mint + verify with replay cache + audit log.

Spec: auth-design.md (composite identity, TTL caches, audit-log shape).

SPIKE-MODE NOTE: this implementation signs locally with a configured
secret. Production will swap to GCP IAM Credentials `signJwt` for true
service-account-rooted identity. The swap point is `_sign_jwt`.

from __future__ import annotations

import logging
import os
import time
import uuid
from dataclasses import dataclass
from typing import Any

import jwt as pyjwt
from cachetools import TTLCache

_AUDIT_LOG = logging.getLogger("lib.a2a.auth.audit")
_REPLAY_CACHE: TTLCache[str, bool] = TTLCache(maxsize=10_000, ttl=300)


def _allowed_issuers() -> set[str]:
    """Allowlist of issuer identities (SA emails). Populated from peers.yaml in production."""
    raw = os.environ.get(
        "A2A_ALLOWED_ISSUERS",
        "agent-canary-spike@i-for-ai.iam.gserviceaccount.com,autonomousagent-vm-runtime@i-for-ai.iam.gserviceaccount.com",
    )
    return {s.strip() for s in raw.split(",") if s.strip()}


def _self_issuer() -> str:
    return os.environ.get(
        "A2A_SELF_ISSUER",
        "autonomousagent-vm-runtime@i-for-ai.iam.gserviceaccount.com",
    )


def _signing_secret() -> str:
    return os.environ.get("A2A_SIGNING_SECRET", "spike-local-secret-DO-NOT-USE-IN-PROD")


@dataclass(frozen=True)
class AgentIdentity:
    issuer: str
    audience: str
    scope: list[str]
    jti: str


class A2AAuthError(Exception):
    pass


class A2AUnauthenticated(A2AAuthError):
    pass


class A2AReplayDetected(A2AAuthError):
    pass


def mint_token(audience: str, scope: list[str], ttl_seconds: int = 300) -> str:
    """Mint a short-lived JWT for outbound A2A calls."""
    now = int(time.time())
    claims = {
        "iss": _self_issuer(),
        "aud": audience,
        "exp": now + ttl_seconds,
        "iat": now,
        "jti": str(uuid.uuid4()),
        "scope": scope,
    }
    return _sign_jwt(claims)


def _sign_jwt(claims: dict[str, Any]) -> str:
    """SPIKE: HS256 with shared secret. PROD: switch to GCP signJwt."""
    return pyjwt.encode(claims, _signing_secret(), algorithm="HS256")


def verify_token(token: str, expected_audience: str) -> AgentIdentity:
    try:
        claims = pyjwt.decode(
            token,
            _signing_secret(),
            algorithms=["HS256"],
            audience=expected_audience,
        )
    except pyjwt.ExpiredSignatureError as exc:
        _emit_audit_log("a2a.verify", outcome="EXPIRED", error=str(exc))
        raise A2AUnauthenticated("token expired") from exc
    except pyjwt.InvalidAudienceError as exc:
        _emit_audit_log("a2a.verify", outcome="WRONG_AUDIENCE", error=str(exc))
        raise A2AUnauthenticated("wrong audience") from exc
    except pyjwt.PyJWTError as exc:
        _emit_audit_log("a2a.verify", outcome="INVALID", error=str(exc))
        raise A2AUnauthenticated(f"invalid token: {exc}") from exc

    issuer = claims.get("iss", "")
    if issuer not in _allowed_issuers():
        _emit_audit_log("a2a.verify", outcome="UNALLOWLISTED_ISSUER", iss=issuer)
        raise A2AUnauthenticated(f"issuer not allowlisted: {issuer}")

    jti = claims.get("jti", "")
    if jti in _REPLAY_CACHE:
        _emit_audit_log("a2a.verify", outcome="REPLAY", jti=jti)
        raise A2AReplayDetected(f"jti replayed: {jti}")
    _REPLAY_CACHE[jti] = True

    identity = AgentIdentity(
        issuer=issuer,
        audience=claims["aud"],
        scope=claims.get("scope", []),
        jti=jti,
    )
    _emit_audit_log("a2a.verify", outcome="OK", iss=issuer, jti=jti)
    return identity


def _emit_audit_log(event: str, **fields: Any) -> None:
    """Cloud Logging will pick this up as a structured entry."""
    msg = f"{event} " + " ".join(f"{k}={v}" for k, v in fields.items())
    _AUDIT_LOG.info(msg, extra={"event": event, **fields})
```

```python
"""TaskSpec <-> A2A Task bridge.

Mapping:
  SpecStatus       A2A TaskState
  ---------------  -------------
  draft            SUBMITTED
  draft_locked     WORKING
  locked           COMPLETED
  superseded       CANCELED
  errored          FAILED  (alias for any unmapped error state)
"""

from __future__ import annotations

from typing import Any

_SPEC_TO_A2A = {
    "draft": "SUBMITTED",
    "draft_locked": "WORKING",
    "locked": "COMPLETED",
    "superseded": "CANCELED",
}


def bridge_taskspec_status_to_a2a(spec_status: str) -> str:
    return _SPEC_TO_A2A.get(spec_status, "FAILED")


def bridge_inbound_to_taskspec(a2a_task: dict[str, Any], identity: Any) -> dict[str, Any]:
    """Convert an inbound A2A Task into the anchors TaskSpec shape.

    DEBT: this returns a dict, not the real TaskSpec dataclass — Day-7
    scope is the bridge contract. Wiring to anchors._draft_from_intent
    happens in `server.py` Step 4. See spike-plan.md Day 7 failure mode
    for the layering issue (anchors does not expose a public API yet).
    """
    return {
        "a2a_task_id": a2a_task["id"],
        "origin": getattr(identity, "issuer", ""),
        "status": "draft",
        "intent_text": _extract_text(a2a_task.get("message", {})),
    }


def _extract_text(message: dict[str, Any]) -> str:
    parts = message.get("parts", [])
    return " ".join(p.get("text", "") for p in parts if "text" in p)
```

```python
"""AgentCard build + sign + verify.

Spec: A2A v1.0.0 §6 (discovery). JCS RFC 8785 canonicalization applied
before signing to ensure JWS verifies across implementations.
"""

from __future__ import annotations

import time
from typing import Any

import jcs
import jwt as pyjwt

from lib.a2a.auth import _signing_secret, _self_issuer


def build_agent_card() -> dict[str, Any]:
    return {
        "id": _self_issuer(),
        "name": "Hermes A2A spike receiver",
        "version": "0.1.0-spike",
        "capabilities": [
            "message/send",
            "message/stream",
            "tasks/get",
            "tasks/subscribe",
            "tasks/cancel",
        ],
        "security_schemes": ["oauth2", "jwt"],
        "jwks_uri": f"https://www.googleapis.com/service_accounts/v1/jwk/{_self_issuer()}",
    }


def sign_card(card: dict[str, Any], ttl_seconds: int = 3600) -> str:
    """JCS-canonicalize the card, then JWS-sign it. Returns compact JWS."""
    canonical = jcs.canonicalize(card).decode("utf-8")
    now = int(time.time())
    return pyjwt.encode(
        {
            "card": canonical,
            "iat": now,
            "exp": now + ttl_seconds,
            "iss": _self_issuer(),
        },
        _signing_secret(),
        algorithm="HS256",
    )


def verify_card(signed_jws: str) -> dict[str, Any]:
    """Verify JWS + canonical round-trip; return the original card dict."""
    import json

    claims = pyjwt.decode(
        signed_jws, _signing_secret(), algorithms=["HS256"], options={"verify_aud": False}
    )
    return json.loads(claims["card"])
```
