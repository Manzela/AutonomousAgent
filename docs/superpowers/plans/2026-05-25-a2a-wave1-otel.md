# A2A Day 6 — OTel `traceparent` Injection (`lib/a2a/client.py`) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add W3C `traceparent` / `tracestate` header injection to every outbound httpx request in `lib/a2a/client.py` using the OpenTelemetry propagation API.

**Architecture:** Extract a `_make_rpc_request` helper (if not already present) that builds the httpx headers dict; call `opentelemetry.propagate.inject(headers)` into that dict before issuing the request. The OTel SDK is added to the `a2a` extra in `pyproject.toml` — only `opentelemetry-api` is strictly required at runtime; tests use `opentelemetry-sdk` (already a transitive dep, pinned in dev) with `InMemorySpanExporter` to create real active spans without a live collector. The sampled bit is read verbatim from the active span's `TraceFlags` — no force-sampling.

**Tech Stack:** `opentelemetry-api>=1.20`, `opentelemetry-sdk>=1.20` (dev only for tests). Both pinned in `pyproject.toml`. `httpx>=0.27` (already in `a2a` extra). `pytest-asyncio>=0.23` (already in `dev` extra).

```python
"""Tests for OTel traceparent injection in lib/a2a/client.py — Day 6.

Per telemetry-design.md:
  - W3C traceparent header injected on every outbound request when a span is active
  - No traceparent injected when there is no active span
  - tracestate passed through verbatim when present in the propagation context
  - sampled bit respected (TraceFlags.SAMPLED -> flags "01"; not sampled -> "00")

Uses opentelemetry.sdk.trace + InMemorySpanExporter to create real active spans
without a live collector. httpx is mocked at the AsyncClient.post level to capture
the headers dict that would be sent on the wire.
"""
from __future__ import annotations

from typing import Any
from unittest.mock import AsyncMock, MagicMock, patch

import httpx
import pytest
from opentelemetry import trace
from opentelemetry.sdk.trace import TracerProvider
from opentelemetry.sdk.trace.export import SimpleSpanProcessor
from opentelemetry.sdk.trace.export.in_memory_span_exporter import InMemorySpanExporter
from opentelemetry.trace import NonRecordingSpan, SpanContext, TraceFlags

from lib.a2a.client import send_message

# ---------------------------------------------------------------------------
# Helpers
# ---------------------------------------------------------------------------

_BASE = "http://testserver/"
_MSG = {"role": "USER", "parts": [{"text": "hi from otel test"}]}

_SUBMITTED_RESPONSE = {
    "jsonrpc": "2.0",
    "id": "test-id",
    "result": {"id": "task-otel-001", "status": "SUBMITTED"},
}


def _make_provider() -> tuple[TracerProvider, InMemorySpanExporter]:
    """Create a fresh TracerProvider + InMemorySpanExporter pair."""
    exporter = InMemorySpanExporter()
    provider = TracerProvider()
    provider.add_span_processor(SimpleSpanProcessor(exporter))
    return provider, exporter


def _mock_httpx_post(response_json: dict) -> AsyncMock:
    """Build an AsyncMock for httpx.AsyncClient.post that returns a fake Response."""
    mock_response = MagicMock(spec=httpx.Response)
    mock_response.raise_for_status = MagicMock()
    mock_response.json.return_value = response_json
    return AsyncMock(return_value=mock_response)


# ---------------------------------------------------------------------------
# Test 1 — traceparent injected when active span present
# ---------------------------------------------------------------------------


@pytest.mark.asyncio
async def test_traceparent_injected_in_send_message() -> None:
    """When an OTel span is active, traceparent appears in the outbound headers."""
    provider, exporter = _make_provider()
    tracer = provider.get_tracer("test")

    captured_headers: dict[str, str] = {}

    async def _capturing_post(
        url: str,
        *,
        json: dict,
        timeout: float,
        headers: dict | None = None,
        **kwargs: Any,
    ) -> httpx.Response:
        # Capture whatever headers the client built
        captured_headers.update(headers or {})
        mock_response = MagicMock(spec=httpx.Response)
        mock_response.raise_for_status = MagicMock()
        mock_response.json.return_value = _SUBMITTED_RESPONSE
        return mock_response

    with tracer.start_as_current_span("test-span") as span:
        trace_id_hex = format(span.get_span_context().trace_id, "032x")
        span_id_hex = format(span.get_span_context().span_id, "016x")

        with patch("httpx.AsyncClient.post", side_effect=_capturing_post):
            await send_message(_BASE, _MSG)

    assert "traceparent" in captured_headers, (
        f"traceparent header missing from outbound request. "
        f"headers captured: {list(captured_headers)}"
    )
    tp = captured_headers["traceparent"]
    # W3C format: 00-{32hex}-{16hex}-{2hex}
    parts = tp.split("-")
    assert len(parts) == 4, f"traceparent has wrong number of segments: {tp!r}"
    assert parts[0] == "00", f"traceparent version must be '00', got {parts[0]!r}"
    assert parts[1] == trace_id_hex, (
        f"trace_id mismatch: expected {trace_id_hex!r}, got {parts[1]!r}"
    )
    assert parts[2] == span_id_hex, (
        f"span_id mismatch: expected {span_id_hex!r}, got {parts[2]!r}"
    )

async def test_no_traceparent_when_no_active_span() -> None:

    # Ensure we are outside any span context
    token = trace.context_api.attach(trace.context_api.Context())

    captured_headers: dict[str, str] = {}

        url: str,

        json: dict,
        timeout: float,
        headers: dict | None = None,
        **kwargs: Any,

    """When tracestate is present in the propagation context, it passes through."""
    from opentelemetry.sdk.trace import ReadableSpan
    from opentelemetry.trace.propagation.tracecontext import TraceContextTextMapPropagator

    provider, _ = _make_provider()
    tracer = provider.get_tracer("test-ts")

    captured_headers: dict[str, str] = {}

    async def _capturing_post(
        url: str,
        *,
        json: dict,
        timeout: float,
        headers: dict | None = None,
        **kwargs: Any,

        mock_response.raise_for_status = MagicMock()
        mock_response.json.return_value = _SUBMITTED_RESPONSE
        return mock_response

    # Inject a tracestate into the context by starting from a carrier
    propagator = TraceContextTextMapPropagator()
    trace_id = 0xABCDEF1234567890ABCDEF1234567890
    span_id = 0x1234567890ABCDEF
```

- [ ] **Step 2: Extract `_make_rpc_request` helper with header injection**

The current `_call` function builds and sends the envelope but has no headers hook. Add a new `_build_otel_headers` helper and wire it into `_call`.

Add the following function **before** `_build_request`:

```python
def _build_otel_headers() -> dict[str, str]:
    """Return a headers dict with W3C traceparent/tracestate injected from the active span.

    Returns an empty dict when:
      - opentelemetry-api is not installed
      - There is no active span in the current context
      - The active span is a NonRecordingSpan (ALWAYS_OFF sampler, no-op provider)

    The sampled bit is read verbatim from TraceFlags — this function never
    force-samples. Callers must not modify the returned dict after injection.
    """
    if not _OTEL_AVAILABLE:
        return {}

    span = _otel_trace.get_current_span()
    ctx = span.get_span_context()
    # SpanContext.is_valid is False for the NoOpSpan / NonRecordingSpan
    if not ctx.is_valid:
        return {}

    headers: dict[str, str] = {}
    _otel_propagate.inject(headers)
    # inject() may produce an empty dict if the span is a NonRecordingSpan
    # (ALWAYS_OFF sampler returns one but its context is not valid — caught above).
    return headers
```

```python
async def _post_with_retry(
    client: httpx.AsyncClient,
    url: str,
    payload: dict[str, Any],
    timeout: float,
    headers: dict[str, str] | None = None,
) -> httpx.Response:
    """POST with exponential-backoff retry on transient transport errors.

    Only retries on httpx.TransportError (connection refused, reset, etc.).
    4xx / JSON-RPC application errors are NOT retried — they are deterministic.
    """
    delay = _RETRY_BASE_DELAY
    for attempt in range(_MAX_RETRIES):
        try:
            return await client.post(
                url, json=payload, timeout=timeout, headers=headers or {}
            )
        except httpx.TransportError as exc:
            if attempt == _MAX_RETRIES - 1:
                raise
            logger.warning(
                "a2a: transient error (attempt %d/%d): %s — retrying in %.1fs",
                attempt + 1,
                _MAX_RETRIES,
                exc,
                delay,
            )
            await asyncio.sleep(delay)
            delay *= 2
    raise AssertionError("unreachable")  # pragma: no cover
```

- [ ] **Step 4: Wire OTel headers into `_call`**

Update `_call` to call `_build_otel_headers()` and pass the result to `_post_with_retry`:

```python
async def _call(
    client: httpx.AsyncClient,
    peer_url: str,
    method: str,
    params: dict[str, Any],
    timeout: float,
) -> Any:
    """Build envelope, POST with OTel traceparent, decode, raise or return result."""
    payload = _build_request(method, params)
    otel_headers = _build_otel_headers()
    resp = await _post_with_retry(client, peer_url, payload, timeout, headers=otel_headers)
    resp.raise_for_status()
    body = resp.json()
    if "error" in body:
        _raise_for_error(body["error"])
    return body["result"]
```
