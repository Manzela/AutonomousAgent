```python
"""A2A AgentCard — discovery + signing (Day 8).

Per spike-plan.md §Day 8:
  - build_agent_card(our_sa, base_url) -> dict
  - canonicalize_card(card) -> bytes  (RFC 8785 JCS via jcs package)
  - sign_card(card, our_sa) -> dict   (adds 'signature' field)
  - verify_card_signature(card_with_sig, issuer_sa) -> bool

GCP signBlob is called via _call_sign_blob (mockable in tests).
JWKS fetch is via _fetch_public_key_for_sa (mockable in tests).

Per DEFAULTS-ACCEPTED.md Q9: signing-key rotation gated by engineering
on-call in dev/staging; sponsor + privacy officer co-sign in prod.

Spec reference: docs/specification.md §4 (AgentCard), §5 (discovery).
"""
from __future__ import annotations

import base64
import logging
import time
from typing import Any

import httpx
import jcs
from cryptography.hazmat.primitives import hashes, serialization
from cryptography.hazmat.primitives.asymmetric import padding
from cryptography.hazmat.primitives.asymmetric.rsa import RSAPublicKey

logger = logging.getLogger(__name__)

# AgentCard capabilities per spike-plan.md §Day 8 and A2A spec §4.
_CAPABILITIES = ["message_send", "message_stream", "task_get", "task_subscribe"]
_SECURITY_SCHEMES = ["oauth2", "jwt"]
# Card validity window: 24 hours.
_CARD_TTL_SECONDS = 86_400


def build_agent_card(our_sa: str, base_url: str) -> dict[str, Any]:
    """Build an unsigned AgentCard dict for this agent.

    Args:
        our_sa: The GCP service account email of this agent.
            e.g. "agent-a@autonomous-agent-2026.iam.gserviceaccount.com"
        base_url: The publicly reachable base URL of this agent's A2A server.
            e.g. "https://agent-a.example.com" or "http://localhost:9001"

    Returns:
        Unsigned card dict. Pass to sign_card to add the 'signature' field.
    """
    now = int(time.time())
    jwks_url = f"https://www.googleapis.com/service_accounts/v1/jwk/{our_sa}"
    return {
        "id": our_sa,
        "base_url": base_url.rstrip("/"),
        "capabilities": _CAPABILITIES,
        "security_schemes": _SECURITY_SCHEMES,
        "jwks_url": jwks_url,
        "iat": now,
        "exp": now + _CARD_TTL_SECONDS,
    }


def canonicalize_card(card: dict[str, Any]) -> bytes:
    """Apply RFC 8785 JSON Canonicalization Scheme to the card dict.

    Produces deterministic UTF-8 bytes: keys sorted recursively, no extra
    whitespace. Used as the signing payload for sign_card and as the
    verification payload for verify_card_signature.

    The 'signature' field MUST be excluded before canonicalizing for
    verification (signature itself is not part of the signed payload).

    Args:
        card: The card dict, optionally including a 'signature' field
              (which is excluded from the canonical form).

    Returns:
        UTF-8 encoded bytes of the canonical JSON representation.
    """
    # Exclude 'signature' from the canonical payload — it's not part of
    # what was signed and would make the verification impossible.
    payload = {k: v for k, v in card.items() if k != "signature"}
    return jcs.canonicalize(payload)
```

```python
# ---------------------------------------------------------------------------
# Private helpers — mockable in tests
# ---------------------------------------------------------------------------


def _call_sign_blob(data: bytes, sa_email: str) -> str:
    """Call GCP IAM signBlob to sign `data` with the SA's private key.

    Returns:
        Base64url-encoded signature string (no padding).

    Raises:
        httpx.HTTPStatusError: on non-2xx GCP response.
        httpx.TimeoutException: on request timeout.

    Note: In tests, mock this function to avoid real GCP calls.
    The mock receives `data` (bytes) and must return a base64url string.
    """
    import google.auth
    import google.auth.transport.requests

    credentials, _ = google.auth.default(
        scopes=["https://www.googleapis.com/auth/cloud-platform"]
    )
    auth_request = google.auth.transport.requests.Request()
    credentials.refresh(auth_request)

    encoded_data = base64.b64encode(data).decode()
    url = (
        f"https://iam.googleapis.com/v1/projects/-/serviceAccounts"
        f"/{sa_email}:signBlob"
    )
    resp = httpx.post(
        url,
        json={"payload": encoded_data},
        headers={"Authorization": f"Bearer {credentials.token}"},
        timeout=10.0,
    )
    resp.raise_for_status()
    signed_blob_b64 = resp.json()["signedBlob"]
    # GCP returns standard base64; convert to base64url without padding.
    sig_bytes = base64.b64decode(signed_blob_b64)
    return base64.urlsafe_b64encode(sig_bytes).rstrip(b"=").decode()


def _fetch_public_key_for_sa(sa_email: str) -> RSAPublicKey:
    """Fetch the RSA public key for `sa_email` from Google's JWKS endpoint.

    In tests, mock this to return a cryptography RSAPublicKey object
    without making a network call.
    """
    jwks_url = f"https://www.googleapis.com/service_accounts/v1/jwk/{sa_email}"
    resp = httpx.get(jwks_url, timeout=10.0)
    resp.raise_for_status()
    # Google JWKS: use the first RSA key (spike uses single-key SA).
    from cryptography.hazmat.primitives.asymmetric.rsa import RSAPublicNumbers
    from cryptography.hazmat.backends import default_backend

    jwk = resp.json()["keys"][0]
    # Decode n and e from base64url.
    def _b64url_to_int(s: str) -> int:
        padding_needed = (4 - len(s) % 4) % 4
        data = base64.urlsafe_b64decode(s + "=" * padding_needed)
        return int.from_bytes(data, "big")

    n = _b64url_to_int(jwk["n"])
    e = _b64url_to_int(jwk["e"])
    return RSAPublicNumbers(e, n).public_key(default_backend())


# ---------------------------------------------------------------------------
# Public signing API
# ---------------------------------------------------------------------------


def sign_card(card: dict[str, Any], our_sa: str) -> dict[str, Any]:
    """Sign the canonicalized card with the agent SA's private key via GCP signBlob.

    Args:
        card: Unsigned card dict (from build_agent_card). Must not have a
              'signature' key — if it does, it is stripped before signing.
        our_sa: Agent SA email to sign with.

    Returns:
        New dict identical to `card` but with 'signature' added (base64url).
    """
    unsigned = {k: v for k, v in card.items() if k != "signature"}
    canonical_bytes = canonicalize_card(unsigned)
    signature = _call_sign_blob(canonical_bytes, our_sa)
    return {**unsigned, "signature": signature}


def verify_card_signature(card_with_sig: dict[str, Any], issuer_sa: str) -> bool:
    """Verify the JWS-style signature on a signed AgentCard.

    Args:
        card_with_sig: Signed card dict (must include 'signature' field).
        issuer_sa: SA email of the card issuer (used for JWKS lookup).

    Returns:
        True if the signature is valid and the card is not expired.

    Raises:
        ValueError: If the card is expired (exp < now).
        KeyError: If 'signature' is missing from card_with_sig.
    """
    # 1. Check expiry before verifying the signature.
    exp = card_with_sig.get("exp", 0)
    if exp < int(time.time()):
        raise ValueError(f"AgentCard is expired (exp={exp})")

    sig_b64url: str = card_with_sig["signature"]

    # 2. Reconstruct the canonical bytes (without the signature field).
    canonical_bytes = canonicalize_card(card_with_sig)

    # 3. Fetch the issuer's public key from Google JWKS.
    public_key = _fetch_public_key_for_sa(issuer_sa)

    # 4. Verify the PKCS1v15 / SHA-256 signature.
    try:
        padding_needed = (4 - len(sig_b64url) % 4) % 4
        sig_bytes = base64.urlsafe_b64decode(sig_b64url + "=" * padding_needed)
        public_key.verify(sig_bytes, canonical_bytes, padding.PKCS1v15(), hashes.SHA256())
        return True
    except Exception:
        return False
```

```python
async def agent_card() -> JSONResponse:
    """Serve the signed AgentCard (Day 8).
```

```python
"""A2A inbound message scrubber (Day 9).

Reads regex patterns from config/a2a/scrubber-patterns.yaml and applies them
to all string values in a params dict (recursively). Used by server.py to
redact PHI from inbound A2A message bodies before attaching to spans or logs.

Per spike-plan.md §Day 9 and integration-points.md §10:
  - Apply scrubber at the A2A boundary (server.py), NOT inside scrubber.py.
  - Patterns are compiled once at import time (no per-request I/O).
  - Replacement string is [REDACTED].
  - False positives acceptable; false negatives are not.
"""
from __future__ import annotations

import logging
import pathlib
import re
from typing import Any

import yaml

logger = logging.getLogger(__name__)

_PATTERNS_PATH = pathlib.Path(__file__).parent.parent.parent / "config" / "a2a" / "scrubber-patterns.yaml"
_REPLACEMENT = "[REDACTED]"


def _load_patterns() -> list[re.Pattern]:
    """Load and compile scrubber patterns from YAML at import time."""
    try:
        with open(_PATTERNS_PATH) as fh:
            config = yaml.safe_load(fh)
        patterns = config.get("patterns", [])
        compiled = [re.compile(p) for p in patterns]
        logger.debug("a2a.scrubber: loaded %d patterns from %s", len(compiled), _PATTERNS_PATH)
        return compiled
    except FileNotFoundError:
        logger.warning("a2a.scrubber: patterns file not found at %s — no scrubbing", _PATTERNS_PATH)
        return []
    except Exception as exc:
        logger.error("a2a.scrubber: failed to load patterns: %s — no scrubbing", exc)
        return []


_COMPILED_PATTERNS: list[re.Pattern] = _load_patterns()


def _scrub_string(value: str) -> str:
    """Apply all compiled patterns to a single string."""
    for pattern in _COMPILED_PATTERNS:
        value = pattern.sub(_REPLACEMENT, value)
    return value


def _scrub_value(value: Any) -> Any:
    """Recursively scrub a value (dict, list, str, or passthrough)."""
    if isinstance(value, str):
        return _scrub_string(value)
    if isinstance(value, dict):
        return {k: _scrub_value(v) for k, v in value.items()}
    if isinstance(value, list):
        return [_scrub_value(item) for item in value]
    # int, float, bool, None — return as-is.
    return value


def scrub_inbound_params(params: dict[str, Any]) -> dict[str, Any]:
    """Scrub PHI patterns from all string values in inbound A2A message params.

    Args:
        params: The `params` dict from an inbound JSON-RPC envelope. May be
                nested (message → parts → text, context → patient → ...).

    Returns:
        New dict with all string values scrubbed. Original is not mutated.
    """
    return {k: _scrub_value(v) for k, v in params.items()}
```

```bash
#!/usr/bin/env bash
# A2A spike Day 10 — end-to-end canary demo.
#
# What this script does:
#   1. Starts the canary compose stack (deploy/docker-compose.canary.yml).
#   2. Waits for both agent-a and agent-canary to be healthy.
#   3. Sends a message/send from agent-a to agent-canary via JSON-RPC.
#   4. Subscribes to the returned task via POST /subscribe (SSE).
#   5. Verifies Cloud Trace has a span for the request (or uses in-memory
#      fallback if SKIP_CLOUD_TRACE=1 is set).
#   6. Prints PASS or FAIL with a short summary.
#
# Prerequisites:
#   - Docker + Docker Compose v2
#   - deploy/docker-compose.canary.yml exists (from Antigravity AG2 PR)
#   - GCP project autonomous-agent-2026 with Cloud Trace enabled
#   - GOOGLE_APPLICATION_CREDENTIALS or Application Default Credentials
#
# Usage:
#   SKIP_CLOUD_TRACE=1 bash scripts/a2a-e2e-demo.sh   # local / no Cloud Trace
#   bash scripts/a2a-e2e-demo.sh                        # full demo with Cloud Trace

set -euo pipefail

COMPOSE_FILE="${REPO_ROOT}/deploy/docker-compose.canary.yml"
AGENT_A_URL="http://localhost:9001"

GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m'

pass() { echo -e "${GREEN}PASS${NC}: $*"; }
fail() { echo -e "${RED}FAIL${NC}: $*"; exit 1; }
warn() { echo -e "${YELLOW}WARN${NC}: $*"; }

echo ""

if [[ ! -f "${COMPOSE_FILE}" ]]; then
    warn "deploy/docker-compose.canary.yml not found."
    warn "Antigravity AG2 PR may not be merged yet."
    warn "Falling back to stub canary on port 9002 via direct Python."

    echo "Starting stub canary (minimal FastAPI echo agent)..."

import json, sys, threading
from http.server import HTTPServer, BaseHTTPRequestHandler
class H(BaseHTTPRequestHandler):
    def do_POST(self):
        length = int(self.headers.get('Content-Length', 0))
        body = json.loads(self.rfile.read(length))
        result = {'jsonrpc': '2.0', 'id': body.get('id'), 'result': {'id': 'canary-task-001', 'status': 'SUBMITTED'}}

        self.send_response(200)
        self.send_header('Content-Type', 'application/json')
        self.send_header('Content-Length', len(resp))
        self.end_headers()
        self.wfile.write(resp)
    def log_message(self, *args): pass

threading.Thread(target=server.serve_forever, daemon=True).start()

wait_healthy() {
    local url="$1/health"
    local label="$2"
    local elapsed=0
    echo -n "Waiting for ${label} health... "
    until curl -sf "${url}" > /dev/null 2>&1; do
        sleep 2
        elapsed=$((elapsed + 2))
        if [[ ${elapsed} -ge ${MAX_WAIT_SECONDS} ]]; then
            fail "${label} did not become healthy after ${MAX_WAIT_SECONDS}s"
        fi
    done
    echo "healthy"
}

# Canary may not have /health if it's the stub; check port liveness instead.
elapsed=0
echo -n "Waiting for agent-canary liveness on :9002... "
until curl -sf "${AGENT_CANARY_URL}/" > /dev/null 2>&1 || \
      nc -z localhost 9002 2>/dev/null; do

echo "Response: ${SEND_RESPONSE}"
TASK_ID=$(echo "${SEND_RESPONSE}" | python3 -c "import json,sys; print(json.load(sys.stdin)['result']['id'])")
TASK_STATUS=$(echo "${SEND_RESPONSE}" | python3 -c "import json,sys; print(json.load(sys.stdin)['result']['status'])")

fi
pass "message/send returned task_id=${TASK_ID} status=${TASK_STATUS}"
echo ""

# --- 4. Subscribe to the task via SSE (POST /subscribe on agent-a) ----------
echo "Step 2: tasks/subscribe via POST /subscribe on agent-a"
SSE_BODY=$(curl -sf -X POST "${AGENT_A_URL}/subscribe" \
    -H "Content-Type: application/json" \
    -d "{\"id\": \"${TASK_ID}\"}" \
    --max-time 15 || true)

SSE_EVENT_COUNT=$(echo "${SSE_BODY}" | grep -c "^data:" || true)
if [[ ${SSE_EVENT_COUNT} -lt 3 ]]; then
    fail "Expected ≥3 SSE events from /subscribe, got ${SSE_EVENT_COUNT}: ${SSE_BODY}"
fi
pass "SSE subscribe emitted ${SSE_EVENT_COUNT} events"
echo ""

# --- 5. Verify AgentCard served by agent-a ----------------------------------
echo "Step 3: GET /.well-known/agent-card.json from agent-a"
CARD_RESPONSE=$(curl -sf "${AGENT_A_URL}/.well-known/agent-card.json")
CARD_ID=$(echo "${CARD_RESPONSE}" | python3 -c "import json,sys; print(json.load(sys.stdin).get('id', 'MISSING'))")

if [[ "${CARD_ID}" == "MISSING" ]]; then
    fail "AgentCard missing 'id' field: ${CARD_RESPONSE}"
fi
pass "AgentCard served with id=${CARD_ID}"
echo ""

# --- 6. Cloud Trace verification (optional) ----------------------------------
if [[ "${SKIP_CLOUD_TRACE}" == "1" ]]; then
    warn "SKIP_CLOUD_TRACE=1 — skipping Cloud Trace verification"
else
    echo "Step 4: verify Cloud Trace has a span for the A2A request"
    # Wait up to 30 seconds for trace ingestion latency.
    sleep 10

        "resource.type=global AND logName=projects/${GCP_PROJECT}/logs/a2a.server" \
        --project="${GCP_PROJECT}" \
        --limit=5 \
        --format="json" 2>/dev/null | python3 -c "import json,sys; logs=json.load(sys.stdin); print(len(logs))" || echo "0")

    if [[ "${TRACE_COUNT}" -gt 0 ]]; then
        pass "Cloud Logging has ${TRACE_COUNT} a2a.server log entries"
    else
        warn "No Cloud Logging entries found (trace may still be ingesting — check Cloud Trace manually)"
        warn "Project: ${GCP_PROJECT}, filter: resource.type=global AND logName=projects/${GCP_PROJECT}/logs/a2a.server"
    fi
fi

echo ""
echo "=== Demo Summary ==="
echo "  message/send:      PASS (task_id=${TASK_ID})"
echo "  tasks/subscribe:   PASS (${SSE_EVENT_COUNT} SSE events)"
echo "  AgentCard:         PASS (id=${CARD_ID})"
if [[ "${SKIP_CLOUD_TRACE}" != "1" ]]; then
    echo "  Cloud Trace:       see warning above or Cloud Console"
fi
echo ""
pass "A2A spike Day 10 demo complete"
```

```markdown
## What works (spike scope)

- **JSON-RPC 2.0 dispatch** (`POST /`): `message/send`, `-32004` stubs for other methods.

## What is broken on purpose

- **`Allow unauthenticated` Cloud Run**: the spike runs allow-all transport so we can iterate without GCP IAM churn. Auth is enforced at the application layer (JWT guard). - **Single-instance JWT replay cache**: intentionally OOM-proof at spike load but will allow replays across replicas. Documented as `TODO(replay-cache-distributed)` in auth.py.
- **AgentCard JWS fallback to unsigned**: intentional for dev/CI where GCP signBlob is unavailable. The fallback logs a `WARNING` but does not break the server. Remove before prod.
```
